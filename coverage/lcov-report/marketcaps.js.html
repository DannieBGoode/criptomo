function generateCurrencyValueHtml( price, currency ) {
  let symbol = '';
  switch ( currency ) {
  case 'EUR':
    symbol = price + '&nbsp;€';
    break;
  case 'USD':
    symbol = '$' + price;
    break;
  default:
    symbol = price + '&nbsp;' + currency.toUpperCase();
  }
  return symbol;
}
